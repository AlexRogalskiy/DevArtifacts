Make an image black and white

This trick involves the use of a canvas element. For it to work, you will have to open this example in a locally running webserver like apache. It won't work if you open it directly because of the browser security restrictions.

Code

// Define an image element and listen for the load event var img = $('<img>'); var canvas = $('#can')[0], context = canvas.getContext('2d'); img.load(function(){ context.drawImage(this, 0, 0); }); img.attr('src','assets/img/Katy_Perry.jpg'); // When the button is pressed, convert // all the pixels to grayscale: $('#btn').click(function(){ var imageData = context.getImageData(0, 0, canvas.width, canvas.height), px = imageData.data, gray = 0; for (var i = 0, n = px.length; i < n; i += 4) { gray = (px[i] + px[i+1] + px[i+2]) / 3; px[i] = gray; // red px[i+1] = gray; // green px[i+2] = gray; // blue } context.putImageData(imageData, 0, 0); });

HTML

<button id="btn">Make it black and white</button> <br /> <canvas id="can" width="220" height="319" />
