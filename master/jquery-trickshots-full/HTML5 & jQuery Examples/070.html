Simplify JavaScript's date and time functions

JavaScript's date and time functions have much to be desired from. But specialized libraries like moment.js can make things much tolerable:

Code

// Create a new moment object var now = moment(); // Create a moment in the past, using a string date var m = moment("April 1st, 2005", "MMM-DD-YYYY"); // Create a new moment using an array var m = moment([2005, 3, 1]); log('What time is it?', moment().format('HH:mm:ss')); var day = moment().day(); // 5 log('What day of the week is it?', moment.weekdays[day] ); log('What is the current month name?', moment.months[moment().month()] ); log('What time is it in London?', moment.utc().format('HH:mm:ss') ); log('What time is it in Japan?', moment.utc().add('hours',9).format('HH:mm:ss') ); log('The next world cup will be ', moment('June 12th, 2014','MMM DD YYYY').fromNow() ); log('What date will it be 7 days from now?', moment().add('days',7).format('MMMM Do, YYYY') ); // Find the duration between two dates var breakfast = moment('8:32','HH:mm'); var lunch = moment('12:52','HH:mm'); log( moment.duration(lunch - breakfast).humanize() + ' between meals' );
