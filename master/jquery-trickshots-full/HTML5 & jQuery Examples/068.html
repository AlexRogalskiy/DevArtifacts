Render HTML with templates

Writing HTML by hand is cumbersome. Generating it through JavaScript is even more so. This is where template libraries like Mustache.js come into play:

Code

// Get the template string var template = $('#template').text(); // Define a list of people var list = [{ "name" : "Peter Parker", "email" : "pp@gmail.com", "age" : 23 },{ "name" : "The Lizard", "email" : "lizrd@flickr.com", "age" : 44 },{ "name" : "Mary Jane", "email" : "mjay@gmail.com", "age" : 22 }]; var people = $('#people'); // Compile the template for better performance. // This turns it into a JavaScript function. var compiled = Mustache.compile(template); $.each(list, function(){ people.append(compiled(this)); });

HTML

<div id="people"></div> <!-- Store the template inline --> <script type="text/html" id="template"> <ul> <li>Name: {{name}}</li> <li>Email: {{email}}</li> <li>Age: {{age}}</li> </ul> </script>
