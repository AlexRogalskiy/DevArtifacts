Give jQuery a utility belt

jQuery is the best choice to manipulate the dom and listen for events. But it comes short when it comes to manipulating data. For this, we have helper libraries like underscore:

Code

// The variables we will be working on var arr = [53, 2, 4, 6, 7, 13, 77, -23, 6, 11, 401]; var obj = { name: "Peter Griffin", age: 42, weight: "won't tell" }; var func = function(name, message){ log(name, 'says', message || this); }; log('Odd elements:', _.filter( arr, function(num){ return num % 2 == 0; }) ); log('Does it contain 401?', _.contains( arr, 401) ); log('Max number:', _.max(arr) ); log('Sorted:', arr.sort(function(a,b){ return a-b}) ); log('Randomized:', _.shuffle(arr) ); log('Unique values:', _.unique(arr) ); log('---'); log('Object keys:', _.keys(obj) ); log('Values:', _.values(obj) ); log('Pick:', _.pick(obj, 'name', 'age') ); log('Size:', _.size(obj) ); log('---'); // Bind the function to a specific value of this var binded = _.bind(func, 'hi!'); binded('Brian'); binded('Stewie'); // Allow the function to run only once var throttled = _.once(func, 10); var dwarfs = 'Bashful Doc Dopey Grumpy Happy Sleepy Sneezy'.split(' '); for(var i=0; i<7; i++){ throttled(dwarfs[i], 'he is hungry'); } log('---'); // You can combine methods through chaining log('Results:'); _.chain(arr).sortBy(function(val){return val}) .filter(function(val){ return val%2 != 0}) .each(function(val){ log(val) });
